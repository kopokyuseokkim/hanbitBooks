import time
from selenium import webdriver
from selenium.webdriver.common.by import By
import numpy as np
import pandas as pd
from konlpy.tag import Hannanum

options = webdriver.ChromeOptions()
options.add_argument('--headless')
options.add_argument('--no-sandbox')
options.add_argument('--disable-dev-shm-usage')
driver = webdriver.Chrome('chromedriver', options=options)

hannanum = Hannanum()

url = "https://www.president.go.kr/ko/contents_new.php?id=brief" 
driver.get(url)

time.sleep(3)

for i in range(1, 11):
  titleXpath = "/html/body/main/div/main/section/div/div[2]/div/table/tbody/tr[" + str(i) + "]/td[2]/a"
  title = driver.find_element(By.XPATH, titleXpath).text
  nouns = hannanum.nouns(title)
  print(nouns)
