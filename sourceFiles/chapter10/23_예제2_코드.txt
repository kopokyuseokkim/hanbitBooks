from numpy import array
from keras.models import Sequential
from keras.layers import Dense, LSTM

xData = []
yData = []

  xData.append([irKorea[i], irKorea[i + 1], irKorea[i + 2]])

for i in range(3, len(irKorea)):
  yData.append(irKorea[i])
